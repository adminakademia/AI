  webui:
    image: ghcr.io/open-webui/open-webui:main

      # Puste / brak = użyj lokalnego Whisper (Local)
      AUDIO_STT_ENGINE: ""

      # Język rozpoznawania – polski
      WHISPER_LANGUAGE: "pl"

      # Model whisper – dobierz do sprzętu:
      # "small" / "medium" / "large-v3" itd.
      WHISPER_MODEL: "small"
    volumes:
     - /docker/openwebui:/app/backend/data
